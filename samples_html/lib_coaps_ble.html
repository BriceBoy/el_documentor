from bleLib import ble_tools, ble_socket
from coapthon.client.helperclient import HelperClient
from coapthon import defines

def create_instance_coaps_ble(ble_name):
    """ Méthode qui retourne une instance de ble_coaps (Utile pour utiliser la bibliothèque avec RobotFramework)

    Args:
        ble_name (string): Nom de l'appareil BLE auquel on veut se connecter (ex: Legrand_pr_D494)

    Returns:
        coaps_ble: Instance de coaps_ble créée
    """
    coaps_ble_inst = coaps_ble(ble_name)
    return coaps_ble_inst

class coaps_ble:

    def __init__(self, ble_name):
        """ Constructeur de la classe coaps_ble. Initialise le ble, créée une socket DTLS et un client CoAPthon

        Args:
            ble_name (string): Nom de l'appareil BLE auquel on veut se connecter (ex: Legrand_pr_D494)
        """
        self.ble = self._connect_to_ble(ble_name)
        self.dtls_cli = self._create_dtls_socket(self.ble)
        self.coap_client = self._create_coap_client(self.dtls_cli)

    def _connect_to_ble(self, ble_name):
        """ Méthode qui permet la connexion à un appareil BLE

        Args:
            ble_name (string): Nom de l'appareil BLE auquel on veut se connecter (ex: Legrand_pr_D494)

        Returns:
            BleTools: Instance de BleTools connectée à l'appareil BLE spécifié
        """
        ble = ble_tools.BleTools()
        ble.init_adapter()

        ble.device_connection(ble_name)
        ble.enable_notification_from_name('UART TX Characteristic')
        ble.change_mtu(250)
        return ble

    def _create_dtls_socket(self, ble):
        """ Méthode qui crée la socket DTLS utilisée pour avec une connexion sécurisée

        Args:
            ble (BleTools): Instance de BleTools

        Returns:
            dtls_cli: Contexte DTLS une fois le handshake réalisé
        """
        args = {
            'ciphers': ['TLS-PSK-WITH-AES-128-GCM-SHA256'],
            'pre_shared_key': ("psk_dtls", '6843363642375366365057513430664544314245694961644144696C76736B36')
        }
        dtls_cli_ctx = ble_socket.init_context_DTLS(debug_level=1, **args)
        my_socket = ble_socket.BLE_socket()
        my_socket.map_to_ble('UART RX Characteristic', 'UART TX Characteristic',
                            ble.write_request_from_name_and_wait_notify, ble.read_request_from_name)
        ble.ble_connect.notify_callback = my_socket.notify_callback
        dtls_cli = dtls_cli_ctx.wrap_socket(my_socket, server_hostname=None, custom_bio=True)
        ble_socket.block(dtls_cli.do_handshake)
        print("Handshake terminé !")
        return dtls_cli

    def _create_coap_client(self, dtls_cli):
        """ Méthode qui crée le client coap

        Args:
            dtls_cli (Contexte DTLS)

        Returns:
            HelperClient: Instance d'HelperClient de CoAPthon permettant de réaliser des requêtes CoAP
        """
        address = ("127.0.0.1", 5683)
        dtls_cli.bind(address) # Important !
        client = HelperClient(server=address, sock=dtls_cli)
        return client

    def close(self):
        """ Méthode qui ferme les connexions CoAP et BLE correctement
        """
        self.coap_client.close()
        self.coap_client.protocol._socket.close()
        self.coap_client.protocol._socket.send(b'') # Nécessaire pour réellement envoyer la trame close_notify
        self.ble.device_disconnection()

    def _convert_code(self, ble_code):
        """ Méthode qui convertit le code BLE en code COAP

        Args:
            ble_code (int): Code de la réponse COAPS en BLE

        Returns:
            int: Code COAP selon la RFC
        """
        getbinary = lambda x, n: format(x, 'b').zfill(n)
        binary = getbinary(ble_code, 8)
        upper_part = str(int(binary[:3], 2)) # Conversion en entier des 3 premiers digits du ble_code binaire
        lower_part = str(int(binary[3:], 2)).zfill(2) # Conversion en entier des 5 derniers digits du ble_code binanire, sur 2 digits ex: 05
        return int(upper_part + lower_part)

    def GET(self, route):
        """ Méthode utilisée pour envoyer une requête GET en CoAP par le BLE

        Args:
            route (string): Route CoAP sur laquelle envoyer la requête

        Returns:
            Array: [response.code, response.payload]
        """
        print("========= GET : " + route + " =========")
        response = self.coap_client.get(route)
        coap_code = self._convert_code(response.code)
        print(f"CODE : {coap_code}\nPAYLOAD : {response.payload}\n")
        return [coap_code, response.payload]

    def POST(self, route, payload, content_type="json"):
        """ Méthode utilisée pour envoyer une requête POST en CoAP par le BLE

        Args:
            route (string): Route CoAP sur laquelle envoyer la requête
            payload (json): Payload de la requête
            content_type (string): Type de contenu a envoyer en payload => [json, cbor, bytes, txt]

        Returns:
            Array: [response.code, response.payload]
        """
        print("========= POST : " + route + " =========")
        if content_type.lower() == "json":
            data = (defines.Content_types["application/json"], payload)
        elif content_type.lower() == "cbor":
            data = (defines.Content_types["application/cbor"], payload)
        elif content_type.lower() == "bytes":
            data = (defines.Content_types["application/octet-stream"], payload)
        elif content_type.lower() == "txt":
            data = (defines.Content_types["text/plain"], payload)
        else:
            return "Error content-type: " + content_type + " not found"
        response = self.coap_client.post(route, data)
        coap_code = self._convert_code(response.code)
        print(f"CODE : {coap_code}\nPAYLOAD : {response.payload}\n")
        return [coap_code, response.payload]

    def DELETE(self, route):
        """ Méthode utilisée pour envoyer une requête DELETE en CoAP par le BLE

        Args:
            route (string): Route CoAP sur laquelle envoyer la requête

        Returns:
            Array: [response.code, response.payload]
        """
        print("========= DELETE : " + route + " =========")
        response = self.coap_client.delete(route)
        coap_code = self._convert_code(response.code)
        print(f"CODE : {coap_code}\nPAYLOAD : {response.payload}\n")
        return [coap_code, response.payload]
